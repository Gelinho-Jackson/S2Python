def max_min(x):
    '''

    :param seq: uma sequencia
    :return: (min, max)

    Entra com um inteiro X, ele retorna o Maior numero e o Minimo.
    '''
    if (len(x) == 0):
        print("Erro, Lista nula ou nao inserida")
        return 0
    else:
        def min_max_calculator(maior, menor, lista, count):
            if (maior>lista[count]):
                maior = lista[count]
            if (menor<lista[count]):
                menor = lista[count]
            if(count == 1):
                return (lista[maior], lista[menor])
            else:
                min_max_calculator(maior, menor, lista, count)
                count-=1
        
    a, b = x[0], x[0]
    a, b = min_max_calculator(a, b, x, (len(x)-1)) 
    return (a, b)


lst = []
for x in range(999):
    lst.append(x)
print(max_min(lst))
